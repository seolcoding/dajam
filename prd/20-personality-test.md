  questionId: string;
  dimension: string;
  direction: 'left' | 'right';  // left = E/S/T/J, right = I/N/F/P
}

function calculateResult(answers: Answer[]): string {
  const scores: Record<string, { left: number; right: number }> = {
    EI: { left: 0, right: 0 },
    SN: { left: 0, right: 0 },
    TF: { left: 0, right: 0 },
    JP: { left: 0, right: 0 },
  };

  answers.forEach(answer => {
    if (answer.direction === 'left') {
      scores[answer.dimension].left++;
    } else {
      scores[answer.dimension].right++;
    }
  });

  const type = [
    scores.EI.left >= scores.EI.right ? 'E' : 'I',
    scores.SN.left >= scores.SN.right ? 'S' : 'N',
    scores.TF.left >= scores.TF.right ? 'T' : 'F',
    scores.JP.left >= scores.JP.right ? 'J' : 'P',
  ].join('');

  return type;  // ì˜ˆ: 'ENFP'
}

function getTypeDistribution(participants: Participant[]): { type: string; count: number }[] {
  const distribution = new Map<string, number>();

  participants.forEach(p => {
    if (p.result) {
